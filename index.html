    <script>

        // 这里可以预设一些书中的经典问答

        const qa_database = [

            { q: "自由", a: "如果你明白‘一切和合的事物都是无常的’，你就被解放了，而不是被某种行为准则所束缚。 [cite: 6]" },

            { q: "孤独", a: "你感到孤单和无聊，所以想要一个伴侣；而你也是自私的，所以想要独立。 " },

            { q: "命运", a: "你有能力去拥有你想要的那种命运，这取决于你自己。因缘既是自由意志又是命运。 [cite: 9]" },

            { q: "善良", a: "修行佛法不仅是为了成为一个好人，更是为了了悟真理。 [cite: 5]" }

        ];


        function ask() {

            const input = document.getElementById('user-input');

            const chatWindow = document.getElementById('chat-window');

            if(!input.value) return;


            // 显示用户问题

            chatWindow.innerHTML += `<div class="message user-msg">${input.value}</div>`;

            

            // 简单的逻辑匹配（实际开发中可对接AI）

            let response = "仁波切曾说：我们总是会陷入想要抓住些什么的圈套中。你可以尝试更深入地理解‘因缘’。 [cite: 6]";

            for(let item of qa_database) {

                if(input.value.includes(item.q)) {

                    response = item.a;

                    break;

                }

            }


            setTimeout(() => {

                chatWindow.innerHTML += `<div class="message bot-msg">${response}</div>`;

                chatWindow.scrollTop = chatWindow.scrollHeight;

            }, 500);


            input.value = '';

        }
